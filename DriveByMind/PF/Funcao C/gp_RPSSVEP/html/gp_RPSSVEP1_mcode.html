1function [GTargets, idx]=gp_RPSSVEP(yr,ys, f_interest, canais, win_type,overlap, NFFT, fs, class_type)
 2% Algoritmo que deteta o alvo (seta ou checkerboard) com base na potência relativa evocada do sinal SSVEP  
 3% Gpires 2015
 4% yr: sinal de referencia sem estimulo (dimensao: NcanaisxTsamples)
 5% ys: sinal SSVEP de resposta a estimululação (dimensao: NcanaisxTsamples)
 6% f_interest: frequencias de interesse (frequencia do estimulo e respetivo harmónico) 
 7% canais: canais usados no algoritmo
 8% win_type: tipo de janela usada no pwelch
 9% overlap: sobreposição do segmento usado no pwelch
10% NFFT: nr de pontos a usar na FFT do pwelch
11% fs: frequencia de amostragem
12% class_type: 1 (Ps - apenas potencia de ys); 2 (Ganho SNR: relação entre Ps e Pr (potencia do sinal e Potencia de baseline))
13
14%alocação de memoria para os parâmetros de entrada
15%inputs
16N=length(canais);
17%Nr de targets (assume sempre que é colocado o harmonico)
18Ntargets=length(f_interest)/2;   
19Pc=zeros(N,NFFT/2);
20Ps=zeros(N,NFFT/2);  %length(f_interest));
21Ps_harm=zeros(N,Ntargets);
22win=zeros(fs,1);
23
24%outputs
25GTargets=zeros(1,Ntargets);
26idx=0;
27
28%escolha da janela para pwelch (terá sempre o tamanho de 1 seg, ou seja, fs amostras)
29%default: janela de hamming
30win_type=2;
31switch win_type
32   case 1
33       win=rectwin(256);  
34   case 2
35        win=hamming(256); %janela de hamming
36   case 3
37       win=hann(256);
38end
39
40k=0;
41if class_type==1
42%lista de canais selecionados
43for i=canais  
44    %i
45    %indice continuo para as matrizes associadas aos canais
46    k=k+1;    
47    [P f] = gp_pwelch(ys(i,:),win,overlap,NFFT,fs);
48    Pc(k,:) = P;
49    for  fi=f_interest
50        Ps(k,fi)=Pc(k,fi)/(Pc(k,fi-1)+Pc(k,fi)+Pc(k,fi+1));
51    end
52    
53    %soma os harmonicos
54    for j=1:Ntargets
55       Ps_harm(k,j)=sum( Ps(k,f_interest([2*j-1:2*j])), 2); 
56    end
57end
58    %soma potencia dos vários canais e devolve potencia para os targets
59    GTargets=sum(Ps_harm,1);
60    %idx devolve o indice do target com maior potência
61    [v idx]=max(GTargets);    
62end
63

68% %potencia com estimulaçao
69% for i=1:length(canais)    %channels
70%     [P f] = pwelch(ys(i,:),256,128,NFFT,fs);
71%     Pc(i,:)=P;
72%     for  fi=f_interest
73%         Ps(i,fi)=Pc(i,fi)/(Pc(i,fi-1)+Pc(i,fi)+Pc(i,fi+1));
74%     end
75%     for j=1:Ntargets
76%          auxT(i,j)=sum( Ps(i,f_interest([2*j-1:2*j])), 2); %soma com harmonico 
77%     end
78% end
79% PsTargets=sum(auxT,1);
80% 
81% %potencia SEM estimulaçao (rest)
82% for i=1:length(canais)    %channels
83%     [P f] = pwelch(yr(i,:),256,128,NFFT,fs);
84%     Pc(i,:)=P;
85%     for  fi=f_interest
86%         Pr(i,fi)=Pc(i,fi)/(Pc(i,fi-1)+Pc(i,fi)+Pc(i,fi+1));
87%     end
88%     for j=1:Ntargets
89%          auxT(i,j)=sum( Pr(i,f_interest([2*j-1:2*j])), 2); %soma com harmonico 
90%       end
91% end
92% PrTargets=sum(auxT,1);
93% 
94% GTargets=PsTargets./PrTargets;
95% end
